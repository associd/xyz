function SelectSort(arr) {
          let array = arr.slice()
          for(let i = 0; i < array.length; i++) {
            for(let j = i; j < array.length; j++) {
              if(array[j] < array[i]) {
                let t = array[j]
                array[j] = array[i]
                array[i] = t
              }
            }
          }
          return array
        }
